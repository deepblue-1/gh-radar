```sql
-- ============================================================
-- Phase 09.1 Plan 01: intraday_upsert_close(jsonb) RPC 생성 (DATA-02 SC #2)
--
-- 결정 근거 (09.1-CONTEXT.md):
--   D-07: partial column UPSERT — open/high/low/trade_amount 미변경 (candle-sync EOD 가 채움)
--   D-20 (NEW): 09:00 첫 cycle 시 stock_daily_ohlcv 오늘자 row 부재 처리
--               INSERT 분기: open=high=low=close=cur_prc 임시 채움 (NOT NULL 제약 충족)
--               UPDATE 분기: high=GREATEST(기존, EXCLUDED.close), low=LEAST(기존, EXCLUDED.close) 단조 갱신
--               EOD candle-sync 가 17:30 에 KRX 공식 OHLC 로 overlay (의도)
--   T-6: REVOKE FROM PUBLIC, anon, authenticated + GRANT TO service_role 명시
--         (Supabase 플랫폼 auto-grant 회귀 방지 — feedback_supabase_rpc_revoke MEMORY 룰)
-- ============================================================

BEGIN;

-- ─────────────────────────────────────────────────────────
-- Step 1. RPC 함수 정의 (SECURITY DEFINER + plpgsql)
--
-- 입력: p_rows = jsonb 배열, 각 원소:
--   { "code": "007460", "date": "2026-05-13",
--     "close": 6760, "volume": 1234567,
--     "change_amount": 100, "change_rate": 1.5 }
--
-- 반환: 영향받은 row 수 (integer)
-- ─────────────────────────────────────────────────────────
CREATE OR REPLACE FUNCTION public.intraday_upsert_close(
  p_rows jsonb
)
RETURNS integer
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  r jsonb;
  affected integer := 0;
BEGIN
  IF p_rows IS NULL OR jsonb_typeof(p_rows) <> 'array' THEN
    RAISE EXCEPTION 'intraday_upsert_close: p_rows must be a jsonb array';
  END IF;

  FOR r IN SELECT * FROM jsonb_array_elements(p_rows)
  LOOP
    INSERT INTO public.stock_daily_ohlcv (
      code,
      date,
      open,
      high,
      low,
      close,
      volume,
      trade_amount,
      change_amount,
      change_rate,
      inserted_at
    )
    VALUES (
      r->>'code',
      (r->>'date')::date,
      (r->>'close')::numeric(20,2),  -- INSERT 시: open = cur_prc (D-20 옵션 D)
      (r->>'close')::numeric(20,2),  -- INSERT 시: high = cur_prc
      (r->>'close')::numeric(20,2),  -- INSERT 시: low  = cur_prc
      (r->>'close')::numeric(20,2),
      COALESCE((r->>'volume')::bigint, 0),
      0,                              -- INSERT 시: trade_amount = 0 (EOD 가 채움)
      NULLIF(r->>'change_amount', '')::numeric(20,2),
      NULLIF(r->>'change_rate',   '')::numeric(10,4),
      now()
    )
    ON CONFLICT (code, date) DO UPDATE
      SET close         = EXCLUDED.close,
          high          = GREATEST(public.stock_daily_ohlcv.high, EXCLUDED.close),
          low           = LEAST(public.stock_daily_ohlcv.low,  EXCLUDED.close),
          volume        = EXCLUDED.volume,
          change_amount = EXCLUDED.change_amount,
          change_rate   = EXCLUDED.change_rate,
          inserted_at   = now();
          -- 의도적 omit: open, trade_amount (D-07 — candle-sync EOD 가 채우는 영역)

    affected := affected + 1;
  END LOOP;

  RETURN affected;
END;
$$;

COMMENT ON FUNCTION public.intraday_upsert_close(jsonb) IS
  'Phase 09.1 intraday-sync: 키움 ka10027 응답을 stock_daily_ohlcv 오늘자 row 에 partial UPSERT. open/trade_amount 미변경. high/low 는 단조 갱신. service_role 만 EXECUTE.';

-- ─────────────────────────────────────────────────────────
-- Step 2. 권한 — REVOKE FROM PUBLIC, anon, authenticated + GRANT TO service_role
--   (T-6 — Supabase 플랫폼 auto-grant 회귀 방지)
-- ─────────────────────────────────────────────────────────
REVOKE EXECUTE ON FUNCTION public.intraday_upsert_close(jsonb) FROM PUBLIC;
REVOKE EXECUTE ON FUNCTION public.intraday_upsert_close(jsonb) FROM anon;
REVOKE EXECUTE ON FUNCTION public.intraday_upsert_close(jsonb) FROM authenticated;

GRANT EXECUTE ON FUNCTION public.intraday_upsert_close(jsonb) TO service_role;

COMMIT;
```
