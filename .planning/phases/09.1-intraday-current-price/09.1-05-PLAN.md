```bash
#!/usr/bin/env bash
set -euo pipefail

# ═══════════════════════════════════════════════════════════════
# setup-intraday-sync-iam.sh
# Phase 09.1 (DATA-02) — intraday-sync 워커의 IAM + VPC + NAT + Static IP + KIWOOM secret
#
# 결정 (09.1-CONTEXT.md):
#   D-12 REVISED: Direct VPC Egress + Cloud NAT + Static External IP (정공법)
#   D-13: runtime SA gh-radar-intraday-sync-sa 신규 + scheduler SA 재사용
#   D-15: KIWOOM_APPKEY/SECRETKEY secret 2개 신설
# ═══════════════════════════════════════════════════════════════

# Section 1: gcloud 가드 (candle-sync mirror)
EXPECTED_PROJECT="${GCP_PROJECT_ID:-}"
EXPECTED_CONFIG="gh-radar"

if [[ -z "$EXPECTED_PROJECT" ]]; then
  echo "ERROR: GCP_PROJECT_ID env var is required" >&2
  echo "Hint: export GCP_PROJECT_ID=gh-radar" >&2
  exit 1
fi

ACTIVE_CONFIG=$(gcloud config configurations list --filter='IS_ACTIVE=true' --format='value(name)')
ACTIVE_PROJECT=$(gcloud config get-value project 2>/dev/null || true)

if [[ "$ACTIVE_CONFIG" != "$EXPECTED_CONFIG" ]]; then
  echo "ERROR: active gcloud configuration is '$ACTIVE_CONFIG', expected '$EXPECTED_CONFIG'" >&2
  exit 1
fi
if [[ "$ACTIVE_PROJECT" != "$EXPECTED_PROJECT" ]]; then
  echo "ERROR: active project is '$ACTIVE_PROJECT', expected '$EXPECTED_PROJECT'" >&2
  exit 1
fi
echo "✓ gcloud guard: config=$ACTIVE_CONFIG, project=$ACTIVE_PROJECT"

# Section 2: API enable (idempotent)
echo "▶ enabling required APIs..."
gcloud services enable \
  run.googleapis.com \
  cloudscheduler.googleapis.com \
  secretmanager.googleapis.com \
  artifactregistry.googleapis.com \
  iam.googleapis.com \
  compute.googleapis.com
echo "✓ APIs enabled (incl. compute.googleapis.com for VPC/NAT)"

# Section 3: VPC + Subnet + Static IP + Router + NAT (idempotent)
REGION=asia-northeast3
VPC_NAME=gh-radar-vpc
SUBNET_NAME=gh-radar-subnet-an3
SUBNET_RANGE=10.10.0.0/26
ROUTER_NAME=gh-radar-router-an3
NAT_NAME=gh-radar-nat-an3
STATIC_IP_NAME=gh-radar-intraday-static-ip

echo "▶ VPC stack (gh-radar-vpc + subnet + router + NAT + static IP)..."

# 3.1 VPC (custom mode)
if ! gcloud compute networks describe "$VPC_NAME" >/dev/null 2>&1; then
  gcloud compute networks create "$VPC_NAME" --subnet-mode=custom
  echo "✓ VPC created: $VPC_NAME"
else
  echo "✓ VPC exists: $VPC_NAME"
fi

# 3.2 Subnet (asia-northeast3, /26 minimum for Direct VPC Egress)
if ! gcloud compute networks subnets describe "$SUBNET_NAME" --region="$REGION" >/dev/null 2>&1; then
  gcloud compute networks subnets create "$SUBNET_NAME" \
    --network="$VPC_NAME" \
    --region="$REGION" \
    --range="$SUBNET_RANGE"
  echo "✓ Subnet created: $SUBNET_NAME ($SUBNET_RANGE)"
else
  echo "✓ Subnet exists: $SUBNET_NAME"
fi

# 3.3 Static External IP reserve
if ! gcloud compute addresses describe "$STATIC_IP_NAME" --region="$REGION" >/dev/null 2>&1; then
  gcloud compute addresses create "$STATIC_IP_NAME" --region="$REGION"
  echo "✓ Static IP reserved: $STATIC_IP_NAME"
else
  echo "✓ Static IP exists: $STATIC_IP_NAME"
fi
STATIC_IP=$(gcloud compute addresses describe "$STATIC_IP_NAME" --region="$REGION" --format='value(address)')

# 3.4 Cloud Router
if ! gcloud compute routers describe "$ROUTER_NAME" --region="$REGION" >/dev/null 2>&1; then
  gcloud compute routers create "$ROUTER_NAME" \
    --network="$VPC_NAME" \
    --region="$REGION"
  echo "✓ Router created: $ROUTER_NAME"
else
  echo "✓ Router exists: $ROUTER_NAME"
fi

# 3.5 Cloud NAT (Static IP 바인딩)
if ! gcloud compute routers nats describe "$NAT_NAME" --router="$ROUTER_NAME" --region="$REGION" >/dev/null 2>&1; then
  gcloud compute routers nats create "$NAT_NAME" \
    --router="$ROUTER_NAME" \
    --region="$REGION" \
    --nat-custom-subnet-ip-ranges="$SUBNET_NAME" \
    --nat-external-ip-pool="$STATIC_IP_NAME"
  echo "✓ Cloud NAT created: $NAT_NAME → Static IP $STATIC_IP"
else
  echo "✓ Cloud NAT exists: $NAT_NAME"
fi

echo "▶ VPC stack ready. Static IP for Kiwoom whitelist: $STATIC_IP"

# Section 4: 선행 SA 존재 확인 — gh-radar-scheduler-sa (Phase 05.1)
for SA in gh-radar-scheduler-sa; do
  SA_EMAIL="${SA}@${EXPECTED_PROJECT}.iam.gserviceaccount.com"
  if ! gcloud iam service-accounts describe "$SA_EMAIL" >/dev/null 2>&1; then
    echo "ERROR: SA '$SA' not found — Phase 05.1 setup-ingestion-iam.sh 가 먼저 실행되어야 함" >&2
    exit 1
  fi
  echo "✓ SA exists (reused): $SA"
done

# Section 5: 신규 intraday-sync runtime SA (idempotent)
INTRADAY_SA_NAME=gh-radar-intraday-sync-sa
INTRADAY_SA_EMAIL="${INTRADAY_SA_NAME}@${EXPECTED_PROJECT}.iam.gserviceaccount.com"

if gcloud iam service-accounts describe "$INTRADAY_SA_EMAIL" >/dev/null 2>&1; then
  echo "✓ SA exists: $INTRADAY_SA_NAME"
else
  gcloud iam service-accounts create "$INTRADAY_SA_NAME" \
    --display-name="gh-radar intraday-sync (Kiwoom ka10027 + Supabase, Phase 09.1 DATA-02)"
  echo "✓ SA created: $INTRADAY_SA_NAME"
fi

# Section 6: KIWOOM Secret 신설 (D-15) — 비어있는 secret, version 은 사용자가 별도 추가
for SECRET in gh-radar-kiwoom-appkey gh-radar-kiwoom-secretkey; do
  if gcloud secrets describe "$SECRET" >/dev/null 2>&1; then
    echo "✓ secret exists: $SECRET"
  else
    gcloud secrets create "$SECRET" --replication-policy=automatic
    echo "✓ secret created: $SECRET (USER ACTION: add version with: echo -n 'VALUE' | gcloud secrets versions add $SECRET --data-file=-)"
  fi
done

# Section 7: 기존 secret 존재 확인 (재사용)
for SECRET in gh-radar-supabase-service-role; do
  if gcloud secrets describe "$SECRET" >/dev/null 2>&1; then
    echo "✓ secret exists (reused): $SECRET"
  else
    echo "ERROR: secret '$SECRET' not found — Phase 05.1 setup-ingestion-iam.sh 가 먼저 실행되어야 함" >&2
    exit 1
  fi
done

# Section 8: Secret accessor 바인딩 (KIWOOM 2 + Supabase 1)
for SECRET in gh-radar-kiwoom-appkey gh-radar-kiwoom-secretkey gh-radar-supabase-service-role; do
  gcloud secrets add-iam-policy-binding "$SECRET" \
    --member="serviceAccount:${INTRADAY_SA_EMAIL}" \
    --role=roles/secretmanager.secretAccessor >/dev/null
  echo "✓ secretAccessor bound: $SECRET → $INTRADAY_SA_NAME"
done

# Section 9: compute.networkUser — Cloud Run Service Agent + Runtime SA 둘 다
# (RESEARCH §4.5 — Direct VPC Egress 전제 조건)
PROJECT_NUMBER=$(gcloud projects describe "$EXPECTED_PROJECT" --format='value(projectNumber)')
RUN_SERVICE_AGENT="service-${PROJECT_NUMBER}@serverless-robot-prod.iam.gserviceaccount.com"

gcloud projects add-iam-policy-binding "$EXPECTED_PROJECT" \
  --member="serviceAccount:${RUN_SERVICE_AGENT}" \
  --role=roles/compute.networkUser \
  --condition=None >/dev/null
echo "✓ compute.networkUser bound: Cloud Run Service Agent"

gcloud projects add-iam-policy-binding "$EXPECTED_PROJECT" \
  --member="serviceAccount:${INTRADAY_SA_EMAIL}" \
  --role=roles/compute.networkUser \
  --condition=None >/dev/null
echo "✓ compute.networkUser bound: $INTRADAY_SA_NAME"

echo ""
echo "✅ setup-intraday-sync-iam.sh complete"
echo ""
echo "🚨 USER ACTIONS REQUIRED 🚨"
echo "  1. Add Secret versions (key 값 채우기):"
echo "     echo -n '<APPKEY>' | gcloud secrets versions add gh-radar-kiwoom-appkey --data-file=-"
echo "     echo -n '<SECRETKEY>' | gcloud secrets versions add gh-radar-kiwoom-secretkey --data-file=-"
echo ""
echo "  2. 키움 IP 화이트리스트에 Static IP 등록:"
echo "     Static IP = $STATIC_IP"
echo "     키움 openapi.kiwoom.com → 마이페이지 → IP 등록 페이지"
echo ""
echo "Next: bash scripts/deploy-intraday-sync.sh"
```

  gcloud scheduler jobs update http "$SCHEDULER_NAME" \
    --location="$REGION" \
    --schedule="$CRON_EXPR" \
    --time-zone="$TIMEZONE" \
    --uri="$SCHEDULER_URI" \
    --http-method=POST \
    --oauth-service-account-email="$SCHEDULER_SA"
else
  echo "▶ creating Cloud Scheduler: $SCHEDULER_NAME"
  gcloud scheduler jobs create http "$SCHEDULER_NAME" \
    --location="$REGION" \
    --schedule="$CRON_EXPR" \
    --time-zone="$TIMEZONE" \
    --uri="$SCHEDULER_URI" \
    --http-method=POST \
    --oauth-service-account-email="$SCHEDULER_SA"

echo "✓ Cloud Scheduler ready: $SCHEDULER_NAME (cron=$CRON_EXPR $TIMEZONE)"

1. **`scripts/smoke-intraday-sync.sh`** 생성:
```bash
#!/usr/bin/env bash
set -euo pipefail

# ═══════════════════════════════════════════════════════════════
# smoke-intraday-sync.sh
# Phase 09.1 (DATA-02) — intraday-sync 정적/원격 검증
#
# INV-1: Cloud Run Job 'gh-radar-intraday-sync' 존재
# INV-2: Cloud Scheduler 'gh-radar-intraday-sync' 존재 + state=ENABLED + cron `* 9-15 * * 1-5`
# INV-3: Static IP 'gh-radar-intraday-static-ip' 존재 + reserved
# INV-4: Subnet 'gh-radar-subnet-an3' 존재 + region=asia-northeast3
# INV-5: Cloud NAT 'gh-radar-nat-an3' 존재 + Static IP 바인딩 확인
# INV-6: Alert policy 'gh-radar-intraday-sync-failure' 존재
#
# 옵션 플래그:
#   --check-cycle    Cloud Run Job manual execute + Supabase rows_affected ≥ MIN_EXPECTED 검증
# ═══════════════════════════════════════════════════════════════

EXPECTED_PROJECT="${GCP_PROJECT_ID:-}"
REGION=asia-northeast3
JOB_NAME=gh-radar-intraday-sync
SCHEDULER_NAME=gh-radar-intraday-sync
STATIC_IP_NAME=gh-radar-intraday-static-ip
SUBNET_NAME=gh-radar-subnet-an3
NAT_NAME=gh-radar-nat-an3
ROUTER_NAME=gh-radar-router-an3
ALERT_NAME=gh-radar-intraday-sync-failure

if [[ -z "$EXPECTED_PROJECT" ]]; then
  echo "ERROR: GCP_PROJECT_ID env var is required" >&2
  exit 1
fi
echo "▶ smoke against project: $EXPECTED_PROJECT"

CHECK_CYCLE=0
for arg in "$@"; do
  case "$arg" in
    --check-cycle) CHECK_CYCLE=1 ;;
    *) echo "Unknown flag: $arg" >&2; exit 1 ;;
  esac
done

PASS=0
FAIL=0

run_check() {
  local name="$1"; shift
  if "$@" >/dev/null 2>&1; then
    echo "  ✓ $name"
    PASS=$((PASS+1))
  else
    echo "  ✗ $name"
    FAIL=$((FAIL+1))
  fi
}

# INV-1: Job
echo "INV-1: Cloud Run Job '$JOB_NAME' 존재"
run_check "Job exists" gcloud run jobs describe "$JOB_NAME" --region="$REGION"

# INV-2: Scheduler + ENABLED + cron
echo "INV-2: Scheduler '$SCHEDULER_NAME' + ENABLED + cron `* 9-15 * * 1-5`"
if gcloud scheduler jobs describe "$SCHEDULER_NAME" --location="$REGION" >/dev/null 2>&1; then
  STATE=$(gcloud scheduler jobs describe "$SCHEDULER_NAME" --location="$REGION" --format='value(state)')
  CRON=$(gcloud scheduler jobs describe "$SCHEDULER_NAME" --location="$REGION" --format='value(schedule)')
  if [[ "$STATE" == "ENABLED" && "$CRON" == "* 9-15 * * 1-5" ]]; then
    echo "  ✓ Scheduler ENABLED + cron OK"
    PASS=$((PASS+1))
  else
    echo "  ✗ Scheduler state=$STATE cron=$CRON"
    FAIL=$((FAIL+1))
  fi
else
  echo "  ✗ Scheduler not found"
  FAIL=$((FAIL+1))
fi

# INV-3: Static IP
echo "INV-3: Static IP '$STATIC_IP_NAME' 존재"
if gcloud compute addresses describe "$STATIC_IP_NAME" --region="$REGION" >/dev/null 2>&1; then
  IP=$(gcloud compute addresses describe "$STATIC_IP_NAME" --region="$REGION" --format='value(address)')
  echo "  ✓ Static IP = $IP (사용자가 키움 화이트리스트 등록 필요)"
  PASS=$((PASS+1))
else
  echo "  ✗ Static IP not found"
  FAIL=$((FAIL+1))
fi

# INV-4: Subnet
echo "INV-4: Subnet '$SUBNET_NAME' 존재 + region=asia-northeast3"
run_check "Subnet exists" gcloud compute networks subnets describe "$SUBNET_NAME" --region="$REGION"

# INV-5: Cloud NAT + Static IP 바인딩
echo "INV-5: Cloud NAT '$NAT_NAME' 존재 + Static IP 바인딩"
if gcloud compute routers nats describe "$NAT_NAME" --router="$ROUTER_NAME" --region="$REGION" >/dev/null 2>&1; then
  NAT_IPS=$(gcloud compute routers nats describe "$NAT_NAME" --router="$ROUTER_NAME" --region="$REGION" --format='value(natIps)')
  if [[ -n "$NAT_IPS" ]]; then
    echo "  ✓ NAT bound to: $NAT_IPS"
    PASS=$((PASS+1))
  else
    echo "  ✗ NAT exists but no natIps bound"
    FAIL=$((FAIL+1))
  fi
else
  echo "  ✗ NAT not found"
  FAIL=$((FAIL+1))
fi

# INV-6: Alert policy
echo "INV-6: Alert policy '$ALERT_NAME' 존재"
if gcloud alpha monitoring policies list --filter="displayName='$ALERT_NAME'" --format='value(name)' | grep -q .; then
  echo "  ✓ alert policy exists"
  PASS=$((PASS+1))
else
  echo "  ✗ alert policy missing"
  FAIL=$((FAIL+1))
fi

# 옵션: --check-cycle (Plan 06 첫 cycle 검증)
if [[ "$CHECK_CYCLE" == "1" ]]; then
  echo ""
  echo "▶ --check-cycle: Cloud Run Job manual execute"
  gcloud run jobs execute "$JOB_NAME" --region="$REGION" --wait
  echo "  ✓ Job execution completed (logs: gcloud logging read \"resource.labels.job_name=$JOB_NAME\" --limit=50)"
fi

echo ""
echo "═══════════════════════════════════"
echo "smoke result: PASS=$PASS, FAIL=$FAIL"
echo "═══════════════════════════════════"
if [[ $FAIL -gt 0 ]]; then
  exit 1
fi
exit 0
```
