```typescript
import type { KiwoomKa10027Row, IntradayCloseUpdate } from "@gh-radar/shared";
import { parseSignedPrice, parseOptionalSignedNumber } from "../kiwoom/parseSigned";

// _AL 접미사 strip + 6자 코드 검증 (RESEARCH §1.5)
function stripAlSuffix(stkCd: string): string {
  return stkCd.replace(/_AL$/, "");
}

/**
 * ka10027 응답 row → IntradayCloseUpdate.
 * 거래정지/휴장 (cur_prc 빈문자열/"0") 는 null 반환 → 호출측 filter.
 */
export function ka10027RowToUpdate(
  row: KiwoomKa10027Row,
  dateIso: string,
): IntradayCloseUpdate | null {
  if (!row.cur_prc || row.cur_prc.trim() === "" || row.cur_prc.trim() === "0") {
    return null;  // RESEARCH §2.3 — 거래정지/휴장
  }

  const code = stripAlSuffix(row.stk_cd);
  if (!/^\d{6}$/.test(code)) {
    throw new Error(`Invalid stk_cd after strip: "${row.stk_cd}" → "${code}"`);
  }

  const close = parseSignedPrice(row.cur_prc).value;
  const changeAmount = parseOptionalSignedNumber(row.pred_pre);
  const changeRate = parseOptionalSignedNumber(row.flu_rt);
  const volume = parseOptionalSignedNumber(row.now_trde_qty) ?? 0;

  return {
    code,
    date: dateIso,
    close,
    changeAmount,
    changeRate,
    volume,
  };
}
```

```typescript
import type { SupabaseClient } from "@supabase/supabase-js";
import type { IntradayCloseUpdate } from "@gh-radar/shared";
import { logger } from "../logger";

const CHUNK_SIZE = 1000;

// IntradayCloseUpdate → RPC 입력 jsonb (snake_case)
type RpcRow = {
  code: string;
  date: string;
  close: number;
  volume: number;
  change_amount: number | null;
  change_rate: number | null;
};

function toRpcRow(u: IntradayCloseUpdate): RpcRow {
  return {
    code: u.code,
    date: u.date,
    close: u.close,
    volume: u.volume,
    change_amount: u.changeAmount,
    change_rate: u.changeRate,
  };
}

export async function upsertIntraday(
  supabase: SupabaseClient,
  updates: IntradayCloseUpdate[],
): Promise<{ rpc_calls: number; rows_affected: number }> {
  if (updates.length === 0) {
    return { rpc_calls: 0, rows_affected: 0 };
  }

  let rpc_calls = 0;
  let rows_affected = 0;

  for (let i = 0; i < updates.length; i += CHUNK_SIZE) {
    const chunk = updates.slice(i, i + CHUNK_SIZE).map(toRpcRow);
    rpc_calls += 1;
    const { data, error } = await supabase.rpc("intraday_upsert_close", {
      p_rows: chunk,
    });
    if (error) {
      logger.error({ error, chunk_index: i, chunk_size: chunk.length }, "intraday_upsert_close failed");
      throw new Error(`intraday_upsert_close RPC failed: ${error.message ?? String(error)}`);
    }
    rows_affected += typeof data === "number" ? data : chunk.length;
  }

  return { rpc_calls, rows_affected };
}
```

3. **`workers/intraday-sync/src/pipeline/bootstrapStocks.ts`** — FK orphan 회피:
```typescript
import type { SupabaseClient } from "@supabase/supabase-js";
import { logger } from "../logger";

const BOOTSTRAP_NAME = "Unknown";
const BOOTSTRAP_MARKET = "KOSPI";  // master-sync 가 다음 실행에서 올바른 market 으로 갱신

/**
 * unique code 를 stocks 에 신규 등록 (ON CONFLICT DO NOTHING).
 * is_delisted=false — 활성 종목 (candle-sync 의 is_delisted=true 와 차이: ka10027 은 살아있는 종목만 응답).
 * master-sync 가 다음 cron 에서 정식 마스터로 갱신.
 */
export async function bootstrapMissingStocks(
  supabase: SupabaseClient,
  codes: string[],
): Promise<{ inserted_or_existing: number }> {
  if (codes.length === 0) {
    return { inserted_or_existing: 0 };
  }

  const unique = Array.from(new Set(codes));
  const rows = unique.map((code) => ({
    code,
    name: BOOTSTRAP_NAME,
    market: BOOTSTRAP_MARKET,
    is_delisted: false,
  }));

  const { error } = await supabase
    .from("stocks")
    .upsert(rows, { onConflict: "code", ignoreDuplicates: true });

  if (error) {
    logger.error({ error, count: unique.length }, "bootstrapMissingStocks failed");
    throw new Error(`bootstrapMissingStocks failed: ${error.message ?? String(error)}`);
  }

  return { inserted_or_existing: unique.length };
}
```

```typescript
import type { Config } from "./config";
import { logger } from "./logger";
import { createKiwoomClient } from "./kiwoom/client";
import { issueKiwoomToken } from "./kiwoom/auth";
import { fetchKa10027 } from "./kiwoom/fetchRanking";
import { ka10027RowToUpdate } from "./pipeline/map";
import { upsertIntraday } from "./pipeline/upsert";
import { bootstrapMissingStocks } from "./pipeline/bootstrapStocks";
import { createSupabaseClient } from "./services/supabase";
import type { IntradayCloseUpdate } from "@gh-radar/shared";

// KST today (YYYY-MM-DD) — UTC 기준 +9h
function todayIsoKst(): string {
  const now = new Date();
  const kst = new Date(now.getTime() + 9 * 60 * 60 * 1000);
  return kst.toISOString().slice(0, 10);
}

function uniqueCodes(updates: IntradayCloseUpdate[]): string[] {
  return Array.from(new Set(updates.map((u) => u.code)));
}

export type CycleResult = {
  duration_ms: number;
  fetched_rows: number;
  skipped_rows: number;
  mapped_rows: number;
  bootstrap_codes: number;
  rpc_calls: number;
  rows_affected: number;
};

export async function runIntradayCycle(config: Config): Promise<CycleResult> {
  const start = Date.now();
  const log = logger.child({ app: "intraday-sync", version: config.appVersion });

  const dateIso = todayIsoKst();
  log.info({ dateIso, kiwoomBaseUrl: config.kiwoomBaseUrl }, "intraday cycle start");

  // 1. Kiwoom client + OAuth token (RESEARCH §5.1 옵션 A — in-memory, per-cycle)
  const kiwoom = createKiwoomClient(config);
  const tk = await issueKiwoomToken(kiwoom, config);
  log.info({ expiresAt: tk.expiresAt.toISOString() }, "kiwoom token issued");

  // 2. ka10027 페이지네이션 호출
  const rows = await fetchKa10027(kiwoom, tk.token, config);
  const fetched_rows = rows.length;
  log.info({ fetched_rows }, "ka10027 fetched");

  // 3. 행수 가드 (RESEARCH §6)
  if (fetched_rows === 0) {
    log.warn("ka10027 empty response — non-trading day or 키움 미응답. exiting normally.");
    return {
      duration_ms: Date.now() - start,
      fetched_rows: 0,
      skipped_rows: 0,
      mapped_rows: 0,
      bootstrap_codes: 0,
      rpc_calls: 0,
      rows_affected: 0,
    };
  }
  if (fetched_rows < config.minExpectedRows) {
    throw new Error(
      `ka10027 returned ${fetched_rows} rows (< ${config.minExpectedRows}) — partial response, aborting cycle`,
    );
  }

  // 4. mapper (null 제거)
  const mapped: IntradayCloseUpdate[] = [];
  let skipped_rows = 0;
  for (const row of rows) {
    const u = ka10027RowToUpdate(row, dateIso);
    if (u === null) {
      skipped_rows += 1;
    } else {
      mapped.push(u);
    }
  }
  log.info({ mapped_rows: mapped.length, skipped_rows }, "ka10027 mapped");

  // 5. Supabase client + bootstrap missing stocks (FK orphan 회피 — RESEARCH §3.4)
  const supabase = createSupabaseClient(config);
  const codes = uniqueCodes(mapped);
  const bootstrap = await bootstrapMissingStocks(supabase, codes);
  log.info({ bootstrap_codes: bootstrap.inserted_or_existing }, "stocks bootstrap complete");

  // 6. RPC upsert (chunked 1000/chunk)
  const upserted = await upsertIntraday(supabase, mapped);
  log.info(
    { rpc_calls: upserted.rpc_calls, rows_affected: upserted.rows_affected },
    "intraday upsert complete",
  );

  const duration_ms = Date.now() - start;
  log.info({ duration_ms }, "intraday cycle complete");

  return {
    duration_ms,
    fetched_rows,
    skipped_rows,
    mapped_rows: mapped.length,
    bootstrap_codes: bootstrap.inserted_or_existing,
    rpc_calls: upserted.rpc_calls,
    rows_affected: upserted.rows_affected,
  };
}
```
