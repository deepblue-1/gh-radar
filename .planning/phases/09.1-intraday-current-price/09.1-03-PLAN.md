import type { AxiosInstance } from "axios";
import type { Config } from "../config";

export type KiwoomToken = {
  token: string;
  tokenType: string;
  expiresAt: Date;
};

export class KiwoomAuthError extends Error {
  constructor(message: string) {
    super(message);
    this.name = "KiwoomAuthError";
  }
}

export function parseKiwoomExpiresDt(s: string): Date {
  if (!/^\d{14}$/.test(s)) {
    throw new Error(`Invalid expires_dt: "${s}" (expected YYYYMMDDhhmmss)`);
  }
  const y  = Number(s.slice(0, 4));
  const mo = Number(s.slice(4, 6)) - 1;
  const d  = Number(s.slice(6, 8));
  const h  = Number(s.slice(8, 10));
  const mi = Number(s.slice(10, 12));
  const se = Number(s.slice(12, 14));
  // KST = UTC+9 → UTC 변환 (RESEARCH §5.2)
  return new Date(Date.UTC(y, mo, d, h - 9, mi, se));
}

export async function issueKiwoomToken(
  client: AxiosInstance,
  config: Config,
): Promise<KiwoomToken> {
  try {
    const res = await client.post("/oauth2/token", {
      grant_type: "client_credentials",
      appkey: config.kiwoomAppkey,
      secretkey: config.kiwoomSecretkey,
    });

    const { token, token_type, expires_dt } = res.data;
    if (!token || typeof token !== "string") {
      throw new KiwoomAuthError(`OAuth response missing token: ${JSON.stringify(res.data)}`);
    }
    if (!expires_dt || typeof expires_dt !== "string") {
      throw new KiwoomAuthError(`OAuth response missing expires_dt: ${JSON.stringify(res.data)}`);
    }

    return {
      token,
      tokenType: token_type ?? "bearer",
      expiresAt: parseKiwoomExpiresDt(expires_dt),
    };
  } catch (err) {
    const status = (err as { response?: { status?: number } }).response?.status;
    if (status === 401) {
      throw new KiwoomAuthError(`Kiwoom OAuth 401 — appkey/secretkey invalid or IP not whitelisted`);
    }
    if (err instanceof KiwoomAuthError) throw err;
    throw err;
  }
}
```

export function parseSignedPrice(s: string | undefined): {
  value: number;
  direction: "up" | "down" | "flat";
} {
  if (!s) throw new Error(`parseSignedPrice: missing signed price`);
  const trimmed = s.trim();
  if (!trimmed) throw new Error(`parseSignedPrice: empty after trim`);

  const sign = trimmed[0];
  const rest = (sign === "+" || sign === "-") ? trimmed.slice(1) : trimmed;
  const abs = Number(rest.replace(/,/g, ""));
  if (!Number.isFinite(abs)) {
    throw new Error(`parseSignedPrice: invalid number "${s}"`);
  }

  const direction: "up" | "down" | "flat" =
    sign === "+" ? "up" :
    sign === "-" ? "down" : "flat";

  return { value: abs, direction };
}

// pred_pre / flu_rt 용 — 부호 보존 (Number 가 leading +/- 자연 처리)
export function parseOptionalSignedNumber(s: string | undefined): number | null {
  if (!s) return null;
  const trimmed = s.trim();
  if (!trimmed) return null;
  const n = Number(trimmed.replace(/,/g, ""));
  return Number.isFinite(n) ? n : null;
}
```

import type { AxiosInstance, AxiosResponse } from "axios";
import type { KiwoomKa10027Row } from "@gh-radar/shared";
import type { Config } from "../config";
import { KiwoomAuthError } from "./auth";
import { logger } from "../logger";

export class KiwoomRateLimitError extends Error {
  constructor(message: string) {
    super(message);
    this.name = "KiwoomRateLimitError";
  }
}

// RESEARCH §1.2 — 사용자 실측 잠금값. 본 body 는 변경하지 않음.
const RKINFO_BODY = {
  mrkt_tp: "000",        // D-02: KOSPI+KOSDAQ 통합
  sort_tp: "1",          // 상승률순
  trde_qty_cnd: "0000",  // 거래량 조건 없음
  stk_cnd: "0",          // 종목조건 전체
  crd_cnd: "0",          // 신용조건 전체
  updown_incls: "1",     // D-03: 등락 미발생 포함
  pric_cnd: "0",         // 가격조건 없음
  trde_prica_cnd: "0",   // 거래대금 조건 없음
} as const;

const RKINFO_PATH = "/api/dostk/rkinfo";

type Ka10027Response = {
  return_code?: number;
  return_msg?: string;
  pred_pre_flu_rt_upper?: KiwoomKa10027Row[];
};

export async function fetchKa10027(
  client: AxiosInstance,
  token: string,
  config: Config,
): Promise<KiwoomKa10027Row[]> {
  const rows: KiwoomKa10027Row[] = [];

  let contYn: "Y" | "N" = "N";   // 첫 호출은 N
  let nextKey: string | undefined;
  let page = 0;

  do {
    page += 1;
    const headers: Record<string, string> = {
      "authorization": `Bearer ${token}`,
      "api-id": "ka10027",
      "cont-yn": contYn === "Y" ? "Y" : "N",
    };
    if (contYn === "Y" && nextKey) {
      headers["next-key"] = nextKey;
    }

    let res: AxiosResponse<Ka10027Response>;
    try {
      res = await client.post<Ka10027Response>(RKINFO_PATH, RKINFO_BODY, { headers });
    } catch (err) {
      const status = (err as { response?: { status?: number } }).response?.status;
      if (status === 401) {
        throw new KiwoomAuthError(`ka10027 401 — token expired or IP not whitelisted`);
      }
      if (status === 429) {
        throw new KiwoomRateLimitError(`ka10027 429 — rate limit exceeded`);
      }
      throw err;
    }

    const page_rows = res.data?.pred_pre_flu_rt_upper ?? [];
    rows.push(...page_rows);

    // 안전상한 (RESEARCH §1.7 + Plan 02 paginationHardCap)
    if (rows.length > config.paginationHardCap) {
      throw new Error(
        `ka10027 pagination exceeded hard cap (${rows.length} > ${config.paginationHardCap}) — possible infinite loop`,
      );
    }

    // 응답 헤더 파싱 (response-headers 는 lowercase)
    const respContYn = (res.headers["cont-yn"] ?? "N").toUpperCase();
    contYn = respContYn === "Y" ? "Y" : "N";
    nextKey = (res.headers["next-key"] ?? "").toString() || undefined;

    logger.debug({ page, page_rows: page_rows.length, total: rows.length, contYn }, "ka10027 page");
  } while (contYn === "Y");

  return rows;
}
