       export interface InfoStockCardProps {
         stock: StockWithProximity;
         /** ⭐ 토글 자리 확보 — Plan 07 에서 WatchlistToggle 주입. 현재는 slot. */

       export function InfoStockCard({ stock, showWatchlistToggle, watchlistToggleSlot }: InfoStockCardProps) {
         const direction: "up" | "down" | "flat" =
           stock.changeRate > 0 ? "up" : stock.changeRate < 0 ? "down" : "flat";
         const colorClass =
           direction === "up" ? "text-[var(--up)]" :
           direction === "down" ? "text-[var(--down)]" : "text-[var(--flat)]";
         const sign = stock.changeRate > 0 ? "+" : "";
         const DirectionIcon = direction === "up" ? ArrowUpRight : direction === "down" ? ArrowDownRight : Minus;

                   {stock.name}
                 </div>
                 <div className="truncate text-[length:var(--t-caption)] text-[var(--muted-fg)] uppercase mono">
                   {stock.market} · {stock.code}
                 </div>
               </div>
               {/* Sparkline */}
               <Sparkline direction={direction} />
               {/* 우측 가격 블록 */}
               <div className="w-24 shrink-0 text-right">

                   <NumberDisplay value={stock.price} format="price" />
                 </div>
                 <div className={cn("mono text-[11px] font-semibold tabular-nums inline-flex items-center gap-0.5", colorClass)}>
                   <DirectionIcon className="size-3" aria-hidden="true" />
