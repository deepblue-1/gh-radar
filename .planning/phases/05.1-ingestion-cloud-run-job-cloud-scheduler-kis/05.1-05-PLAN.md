```bash
# V1: smoke 전체 PASS
bash scripts/smoke-ingestion.sh gh-radar-ingestion asia-northeast3
[ $? -eq 0 ] || { echo "FAIL V1: smoke failed"; exit 1; }
echo "✓ V1 smoke all INV PASS"

# V2 (VALIDATION Row 05.1-05-01): Supabase freshness (거래일) OR 휴장일 로그
: "${SUPABASE_URL:?}"; : "${SUPABASE_SERVICE_ROLE_KEY:?}"
MAX_TS=$(curl -fsS "${SUPABASE_URL}/rest/v1/stocks?select=updated_at&order=updated_at.desc&limit=1" \
  -H "apikey: $SUPABASE_SERVICE_ROLE_KEY" \
  -H "Authorization: Bearer $SUPABASE_SERVICE_ROLE_KEY" | jq -r '.[0].updated_at')
echo "  stocks.max(updated_at) = $MAX_TS"

# V3 (Row 05.1-05-02): 휴장일 로그 존재 (거래일이면 skip)
HOLIDAY_HITS=$(gcloud logging read \
  'resource.type="cloud_run_job" AND resource.labels.job_name="gh-radar-ingestion" AND jsonPayload.msg="non-trading day detected, exiting"' \
  --freshness=10m --limit=1 --format='value(timestamp)' | wc -l | tr -d ' ')
echo "  non-trading day logs (10m) = $HOLIDAY_HITS"

# V4 (Row 05.1-05-03): cycle complete 로그 (거래일이면 존재)
CYCLE_HITS=$(gcloud logging read \
  'resource.type="cloud_run_job" AND resource.labels.job_name="gh-radar-ingestion" AND jsonPayload.msg="cycle complete"' \
  --freshness=5m --limit=1 --format='value(timestamp)' | wc -l | tr -d ' ')
echo "  cycle complete logs (5m) = $CYCLE_HITS"

# V5 (Row 05.1-05-04): 실패/EGW00201 0건 (의도 실패 테스트는 별도 시간창)
FAIL_HITS=$(gcloud logging read \
  'resource.type="cloud_run_job" AND resource.labels.job_name="gh-radar-ingestion" AND (jsonPayload.msg="cycle failed" OR textPayload:"EGW00201")' \
  --freshness=5m --limit=1 --format='value(timestamp)' | wc -l | tr -d ' ')
[ "$FAIL_HITS" = "0" ] || { echo "FAIL V5: cycle failed/EGW00201 detected in 5m window"; exit 1; }
echo "✓ V5 no cycle failed / EGW00201"

# V6: 거래일 OR 휴장일 — 둘 중 하나의 증거 필수
if [ "$CYCLE_HITS" -gt 0 ] || [ "$HOLIDAY_HITS" -gt 0 ]; then
  echo "✓ V6 cycle evidence present"
else
  echo "FAIL V6: neither cycle complete nor non-trading day log in recent window"; exit 1
fi

# V7: Scheduler ENABLED 유지
STATE=$(gcloud scheduler jobs describe gh-radar-ingestion-scheduler --location=asia-northeast3 --format='value(state)')
[ "$STATE" = "ENABLED" ] || { echo "FAIL V7: scheduler state=$STATE"; exit 1; }
echo "✓ V7 scheduler ENABLED"

# V8 (Manual): 알림 이메일 수신 시각은 사용자가 보고하므로 Claude는 "manual_verified=true" flag만 요구
echo ""
echo "✅ Wave 4 complete — all INVs PASS, Scheduler live."
echo "   Manual verification (email receipt) 결과를 SUMMARY에 기록 필요."
```
