```ts
import { test, expect } from '@playwright/test';
import { mockDiscussionsApi } from '../fixtures/mock-api';
import type { Discussion } from '@gh-radar/shared';

const CODE = '005930';

function makeRow(partial: Partial<Discussion>): Discussion {
  return {
    id: partial.id ?? 'id-' + Math.random(),
    stockCode: CODE,
    postId: partial.postId ?? 'p-' + Math.random(),
    title: partial.title ?? '샘플',
    body: partial.body ?? null,
    author: partial.author ?? 'tester',
    postedAt: partial.postedAt ?? new Date().toISOString(),
    scrapedAt: partial.scrapedAt ?? new Date().toISOString(),
    url: 'https://stock.naver.com/domestic/stock/005930/discussion/x?chip=all',
    relevance: partial.relevance ?? null,
    classifiedAt: partial.classifiedAt ?? null,
  };
}

const ROWS: Discussion[] = [
  makeRow({ id: '1', title: '실적 발표 후 급등 이유', relevance: 'price_reason' }),
  makeRow({ id: '2', title: '2차전지 테마 로테이션', relevance: 'theme' }),
  makeRow({ id: '3', title: 'ㅋㅋㅋ뇌피셜', relevance: 'noise' }),
  makeRow({ id: '4', title: '공시 내용 요약', relevance: 'news_info' }),
];

test.describe('Phase 08.1 discussion filter toggle', () => {
  test('기본 ON(meaningful) — noise 제외', async ({ page }) => {
    await mockDiscussionsApi(page, CODE, ROWS);
    await page.goto(`/stocks/${CODE}/discussions`);
    await expect(page.getByTestId('discussion-filter-toggle')).toBeVisible();
    // 토글 checked
    const toggle = page.getByRole('switch', { name: /의미있는 토론만 보기/ });
    await expect(toggle).toBeChecked();
    // 리스트에 noise 행 없음
    await expect(page.getByText('ㅋㅋㅋ뇌피셜')).toHaveCount(0);
    await expect(page.getByText('실적 발표 후 급등 이유')).toBeVisible();
  });

  test('토글 OFF → filter=all URL + noise 포함', async ({ page }) => {
    await mockDiscussionsApi(page, CODE, ROWS);
    await page.goto(`/stocks/${CODE}/discussions`);
    const toggle = page.getByRole('switch', { name: /의미있는 토론만 보기/ });
    await toggle.click();
    await expect(page).toHaveURL(/filter=all/);
    await expect(page.getByText('ㅋㅋㅋ뇌피셜')).toBeVisible();
  });

  test('직접 ?filter=all 진입 — 토글 OFF 상태', async ({ page }) => {
    await mockDiscussionsApi(page, CODE, ROWS);
    await page.goto(`/stocks/${CODE}/discussions?filter=all`);
    const toggle = page.getByRole('switch', { name: /의미있는 토론만 보기/ });
    await expect(toggle).not.toBeChecked();
    await expect(page.getByText('ㅋㅋㅋ뇌피셜')).toBeVisible();
  });

  test('filter=meaningful 빈 배열 → 전용 카피', async ({ page }) => {
    await mockDiscussionsApi(page, CODE, []);
    await page.goto(`/stocks/${CODE}/discussions?filter=meaningful`);
    await expect(page.getByTestId('discussion-page-empty')).toContainText(
      '의미있는 토론이 아직 없어요. 토글을 꺼서 전체 글을 볼 수 있어요.'
    );
  });
});
```

```sql
-- 2) 최근 1시간 수집분의 분류 속도

  count(*) FILTER (WHERE classified_at IS NULL) AS pending,
  count(*) FILTER (WHERE classified_at IS NOT NULL) AS classified,
  min(scraped_at) AS oldest_scrape
```
