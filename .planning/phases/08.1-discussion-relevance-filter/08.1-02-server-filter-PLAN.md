```ts
import { describe, it, expect, vi } from 'vitest';
import request from 'supertest';
import { createApp } from '../../app.js';

describe('GET /api/stocks/:code/discussions (Phase 08.1 filter)', () => {
  const mockSupabase = {
    from: vi.fn(() => mockSupabase),
    select: vi.fn(() => mockSupabase),
    eq: vi.fn(() => mockSupabase),
    gte: vi.fn(() => mockSupabase),
    lt: vi.fn(() => mockSupabase),
    or: vi.fn(() => mockSupabase),
    order: vi.fn(() => mockSupabase),
    limit: vi.fn(() => mockSupabase),
    maybeSingle: vi.fn(async () => ({ data: { code: '005930' }, error: null })),
    then: undefined as never,
  };
  // supabase.from('discussions').select().eq().gte()... 의 최종 await 은
  // { data: [...], error: null } 을 반환하도록 `limit` mock 에서 설정.

  it('filter=meaningful 이면 or() 체인이 호출된다', async () => {
    // rows 에 relevance='noise' 1건, 'price_reason' 1건, null 1건을 mock
    // 핸들러가 filter=meaningful 일 때만 or() 호출
    const app = createApp({ supabase: mockSupabase as any });
    const res = await request(app).get('/api/stocks/005930/discussions?filter=meaningful&days=1&limit=50');
    expect(res.status).toBe(200);
    expect(mockSupabase.or).toHaveBeenCalledWith('relevance.is.null,relevance.neq.noise');
  });

  it('filter=all (또는 미지정) 이면 or() 체인이 호출되지 않는다', async () => {
    const app = createApp({ supabase: mockSupabase as any });
    mockSupabase.or.mockClear();
    await request(app).get('/api/stocks/005930/discussions?days=1&limit=50');
    expect(mockSupabase.or).not.toHaveBeenCalled();
  });

  it('filter=foo 은 400 을 반환한다', async () => {
    const app = createApp({ supabase: mockSupabase as any });
    const res = await request(app).get('/api/stocks/005930/discussions?filter=foo');
    expect(res.status).toBe(400);
  });
});
```

NOTE: 실제 mock 구조는 기존 server test 의 헬퍼(예: `createMockSupabase`)가 있으면 재사용. 위 샘플은 최소 윤곽 — 실제 구현 시 기존 테스트 파일 스타일을 literal 하게 follow.
