       ```ts
       import type { Page, Route } from '@playwright/test';
       import type { Stock } from '@gh-radar/shared';
       import { FIXTURE_SAMSUNG } from './stocks';

       /**
        * playwright 테스트용 API 모킹.
        * - /api/stocks/search?q=... → 검색 결과 (쿼리별 분기)
        * - /api/stocks/:code → 단일 종목 또는 404
        */
       export async function mockStockApi(page: Page, opts: {
         searchResults?: Stock[];
         detailByCode?: Record<string, Stock>;
         detailStatusByCode?: Record<string, number>;
       } = {}) {
         const {
           searchResults = [FIXTURE_SAMSUNG],
           detailByCode = { '005930': FIXTURE_SAMSUNG },
           detailStatusByCode = {},
         } = opts;

         // /api/stocks/search?q=...
         await page.route('**/api/stocks/search*', async (route: Route) => {
           await route.fulfill({
             status: 200,
             contentType: 'application/json',
             headers: { 'x-request-id': 'test-req-id' },
             body: JSON.stringify(searchResults),
           });
         });

         // /api/stocks/:code (search 가 아닌 것만)
         await page.route(/\/api\/stocks\/([A-Za-z0-9]{1,10})(?!\/)$/, async (route: Route) => {
           const url = route.request().url();
           const match = url.match(/\/api\/stocks\/([A-Za-z0-9]{1,10})/);
           const code = match?.[1] ?? '';
           const status = detailStatusByCode[code];
           if (status && status >= 400) {
             await route.fulfill({
               status,
               contentType: 'application/json',
               body: JSON.stringify({ error: { code: 'STOCK_NOT_FOUND', message: `stock ${code} not found` } }),
             });
             return;
           }
           const stock = detailByCode[code];
           if (!stock) {
             await route.fulfill({
               status: 404,
               contentType: 'application/json',
               body: JSON.stringify({ error: { code: 'STOCK_NOT_FOUND', message: `stock ${code} not found` } }),
             });
             return;
           }
           await route.fulfill({
             status: 200,
             contentType: 'application/json',
             headers: { 'x-request-id': 'test-req-id' },
             body: JSON.stringify(stock),
           });
         });

         // 다른 API (/api/scanner 등) 는 그대로 통과 — webServer 가 백엔드 없으면 실패할 수 있으므로 /scanner 페이지가 필요한 테스트는 스캐너 API 도 모킹
         await page.route('**/api/scanner*', async (route) => {
           await route.fulfill({
             status: 200,
             contentType: 'application/json',
             headers: { 'x-last-updated-at': new Date().toISOString() },
             body: JSON.stringify([]),
           });
         });
       }
       ```

       ```ts
       import { test, expect } from '@playwright/test';
       import { mockStockApi } from '../fixtures/mock-api';

       test.describe('Phase 6 — 전역 검색 (SRCH-01/02)', () => {
         test.beforeEach(async ({ page }) => {
           await mockStockApi(page, { searchResults: [FIXTURE_SAMSUNG] });
         });

         test('⌘K 단축키 → 검색 → 선택 → /stocks/005930 이동', async ({ page }) => {
           await page.goto('/scanner');
           // mod+k (playwright 의 Meta = Mac Cmd, Control = Win/Linux Ctrl — 한 번에 둘 다 통과하는 것은 없음)
           const isMac = process.platform === 'darwin';
           await page.keyboard.press(isMac ? 'Meta+K' : 'Control+K');
           await expect(page.getByRole('dialog')).toBeVisible();
           await page.getByPlaceholder('종목명 또는 종목코드를 입력하세요').fill('삼성');
           // 300ms debounce 대기
           await expect(page.getByRole('option', { name: /삼성전자/ })).toBeVisible({ timeout: 2000 });
           await page.getByRole('option', { name: /삼성전자/ }).click();
           await expect(page).toHaveURL(/\/stocks\/005930$/);
         });

         test('헤더 트리거 클릭으로도 Dialog 오픈', async ({ page }) => {
           await page.goto('/scanner');
           await page.getByRole('button', { name: '종목 검색 열기' }).first().click();
           await expect(page.getByRole('dialog')).toBeVisible();
         });

         test('빈 결과 카피 노출', async ({ page }) => {
           await mockStockApi(page, { searchResults: [] });
           await page.goto('/scanner');
           await page.keyboard.press(process.platform === 'darwin' ? 'Meta+K' : 'Control+K');
           await page.getByPlaceholder('종목명 또는 종목코드를 입력하세요').fill('xyz');
           await expect(page.getByText(/"xyz" 에 해당하는 종목이 없습니다/)).toBeVisible({ timeout: 2000 });
         });
       });
       ```

       ```ts
       import { test, expect } from '@playwright/test';
       import AxeBuilder from '@axe-core/playwright';
       import { mockStockApi } from '../fixtures/mock-api';

       test.describe('Phase 6 — 접근성 (axe)', () => {
         test.beforeEach(async ({ page }) => {
           await mockStockApi(page);
         });

         test('/stocks/005930 — critical/serious 위반 0', async ({ page }) => {
           await page.goto('/stocks/005930');
           await page.getByRole('heading', { name: '삼성전자' }).waitFor();
           const results = await new AxeBuilder({ page })
             .withTags(['wcag2a', 'wcag2aa'])
             .analyze();
           const blocking = results.violations.filter(
             (v) => v.impact === 'critical' || v.impact === 'serious',
           );
           expect(blocking, JSON.stringify(blocking, null, 2)).toEqual([]);
         });

         test('⌘K Dialog 열린 상태 — critical/serious 위반 0', async ({ page }) => {
           await page.goto('/scanner');
           const isMac = process.platform === 'darwin';
           await page.keyboard.press(isMac ? 'Meta+K' : 'Control+K');
           await expect(page.getByRole('dialog')).toBeVisible();
           const results = await new AxeBuilder({ page })
             .include('[role="dialog"]')
             .withTags(['wcag2a', 'wcag2aa'])
             .analyze();
           const blocking = results.violations.filter(
             (v) => v.impact === 'critical' || v.impact === 'serious',
       ```
